# Dashboard em Python - Covid19

# Dataset: https://covid.saude.gov.br
# Bibliotecas: Plotly/Dash e Dash BootStrap Components

# Crie o diretório do projeto

# Crie o ambiente virtual no powershell
python -m venv venv
.\venv\Scripts\Activate.ps1

# Instale as seguintes bibliotecas:
pip install dash
pip install dash-core-components
pip install dash-html-components
pip install dash-bootstrap-components
pip install dash-components-library
pip install plotly-express
pip install numpy
pip install pandas

# Para rodar o projeto: python dashboard.py
